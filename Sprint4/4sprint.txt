select 
	ts,
	case
		when description = 'rain' then 'bad'
		when description = 'snow' then 'bad'
		else 'good'
	end weather_conditions
from 
	weather_records;  /*разделили на две группы*/

	ts,
	case
		when description like '%rain%' or description like '%storm%' then 'bad'
		else 'good'
	end weather_conditions
from 
	weather_records;
where 
	ts between '2017-11-05' AND '2017-11-06'; /*сделали выборку за период*/

select 
	count(*) as trips_amount,
	cabs.company_name
from
	trips
inner join 
	cabs on trips.cab_id = cabs.cab_id
where 
	start_ts >= '2017-11-15' and
	start_ts::Date <= '2017-11-16'
group by cabs.company_name
order by trips_amount desc;
